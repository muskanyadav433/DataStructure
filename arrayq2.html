//C program to enter an array from user and print the array in reverse order ?

        #include
        
        int main(){   
        int arr[10],i;
        
        printf("You can only enter upto 10 values\n");
        
        printf("Enter the elements in Array\n");
        
        for( i=0;i<=9;i++)
        {
        printf("%d Element:",i+1);
        
        scanf("%d",&arr[i]);
        
        printf("\n");
        }
        
        printf("The Array in reverse order is given below\n");
        for( i=9;i>=0;i--)
        printf("%d\t",arr[i]);
        return 0;
        }
